最佳答案
设表名为TABLENAME,使用一条查询语句实现：
SELECT 品名,数量=sum(数量),封皮颜色=dbo.getcol(品名) from tablename group by 品名

但要先建立一个列转行的函数：
create FUNCTION getcol
(@PM VARCHAR(50),--品名
@Deli varchar(1)=',')--可选间隔符，默认为逗点。
RETURNS varchar(1000)
AS
BEGIN
if @deli=''
set @deli=' '
declare @f1 varchar(100),@result varchar(1000)
set @result=''

declare ddcursor cursor for SELECT DISTINCT 封皮颜色 FROM TABLENAME WHERE 品名=@PM
open ddcursor
fetch next from ddcursor into @f1
while @@fetch_status!=-1
begin
set @result=@result+@deli+isnull(@f1,'')
fetch next from ddcursor into @f1
end
close ddcursor
deallocate ddcursor
set @result=substring(@result,2,1000)

return (@result)
end
GO

如果能告诉我每个品名最多多少种颜色，一条语句也能实现。
＝＝＝＝＝＝＝＝＝＝＝＝＝＝

已知最大颜色数时用一条语句：

假设每品名最多有三种颜色按下面这样，如果有更多，请自己照样增加LEFT JOIN语句就行了。
SELECT d.品名,数量=SL1+isnull(SL2,0)+isnull(SL3,0),封皮颜色=FP1+isnull(FP2,'')+isnull(FP3,'') from
(select 品名,sl1=数量,fp1=封皮颜色 from
(SELECT *,SN=(SELECT COUNT(1) FROM TABLENAME A WHERE A.品名=b.品名 and a.编号<b.编号) tablename b) c where sn=1) d
left join
(select 品名,sl2=数量,fp2=','+封皮颜色 from
(SELECT *,SN=(SELECT COUNT(1) FROM TABLENAME A WHERE A.品名=b.品名 and a.编号<b.编号) tablename b) c where sn=2) e
on e.品名=d.品名
left join
(select 品名,sl3=数量,fp3=','+封皮颜色 from
(SELECT *,SN=(SELECT COUNT(1) FROM TABLENAME A WHERE A.品名=b.品名 and a.编号<b.编号) tablename b) c where sn=3) f
on f.品名=e.品名
