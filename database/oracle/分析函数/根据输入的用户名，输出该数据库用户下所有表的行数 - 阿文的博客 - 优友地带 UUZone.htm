根据输入的用户名，输出该数据库用户下所有表的行数

--创建临时表
CREATE TABLE tempcount
(
COUNT      NUMBER(10),   --行数                 
TABLE_NAME VARCHAR2(50)  --表名
)

--当然该用户要用访问其它用户表的权限
CREATE OR REPLACE PROCEDURE p_user_table_count
--功能：根据输入的用户名，输出该数据库用户下所有表的行数
--设计：阿文
--编码：阿文
--日期：2006年07月05日
(
iuser       VARCHAR2               --输入数据库用户名
)
AS
  vuser   VARCHAR2(20):=upper(iuser);--将输入的用户名大写
  vnum    NUMBER(10);                --数字变量
  vsql    VARCHAR2(200);             --动态SQL语句
  vcount  INT:=1;                    --记数器
BEGIN
  DBMS_OUTPUT.ENABLE(1000000);
  --先判断该数据库下有没有该用户
  SELECT COUNT(*) INTO vnum FROM all_users t WHERE t.username=vuser;
  IF vnum=0 THEN     --如果该数据库下没有该用户，报表错提示信息
     DBMS_OUTPUT.put_line(chr(13)||'EXECUTE ERROR!用户'||vuser||'不存在！');
  ELSE               --如果该数据库有该用户
     --判断该用户有没有建立表
     SELECT COUNT(*) INTO vnum FROM All_Tables t WHERE t.owner=vuser;
      IF vnum=0 THEN --如果该用户没有建立表
         DBMS_OUTPUT.put_line(chr(13)||'用户 '||vuser||' 没有建立任何表！');
      ELSE           --如果该用户有建立表,则循环输出该用户下每张表的行数
      DBMS_OUTPUT.put_line(chr(13)||'---统计用户 '||vuser||' 各表的行数,共'||vnum||'张表---');
      --隐性游标c
      FOR c IN (SELECT t.table_name FROM All_Tables t WHERE t.owner=vuser)
      LOOP
         DELETE FROM tempcount;  --清空临时表
         --动态SQL语句,将行数插入到临时表
         vsql:='INSERT INTO firefly.tempcount SELECT COUNT(*),'''||vuser||''' FROM '||vuser||'.'||c.table_name;
         --执行该动态SQL语句
         EXECUTE IMMEDIATE vsql;
         --取出该表的行数赋给变量
         SELECT p.count INTO vnum FROM firefly.tempcount p;
         --格式化输出
         DBMS_OUTPUT.put_line(to_char(vcount,'999')||' '||'表'||rpad(c.table_name,20,' ')||' 共有'||to_char(vnum,'999999')||'行');
         vcount:=vcount+1;
      END LOOP;
         DBMS_OUTPUT.put_line('------------------------------------------');
      END IF;
  END IF;
 
  --提交
  COMMIT;
 
EXCEPTION            --异常处理
  WHEN OTHERS THEN   --输出错误代码及信息
   DBMS_OUTPUT.put_line('EXECUTE FIAL! '||Sqlcode);
   DBMS_OUTPUT.put_line(substr(Sqlerrm,1,200));
END p_user_table_count;
